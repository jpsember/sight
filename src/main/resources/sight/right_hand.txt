% This is causing formatting trouble.  Best to work with a larger page size, as
% we are extracting much smaller rects from it anways.

% #(set-default-paper-size '(cons (* 75 mm) (* 22 mm)))

\paper {
  horizontal-shift = -2.0\mm
}

\score {
   \new PianoStaff <<
    \new Staff   <<
      {

        % This seems to make the notes appear at specific horizontal intervals;
        % See https://lilypond.org/doc/v2.23/Documentation/snippets/spacing
        %
        \override Score.SpacingSpanner.strict-note-spacing = ##t
        % Not sure why, but I increased this from 1/16 and the clef/first note collision goes away
        \set Score.proportionalNotationDuration = #(ly:make-moment 2/16)

        \key [!key]
        \clef treble

        %\time 4/4          % Specify time sig
        % \omit Staff.Clef
        \omit Staff.TimeSignature
        \omit Score.BarLine  % hide the bar lines

        [!notes]

      }

    >>
   >>
}

\header {
  tagline = ""  % removed
}
